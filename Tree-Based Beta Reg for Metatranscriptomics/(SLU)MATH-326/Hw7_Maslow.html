# need to find an alpha and beta that simultaneously have alpha/(alpha+beta) = 0.3

mean = 0.3   # set a target for the mean = alpha/(alpha+beta) of our beta distribution
  # set up our probability statement
target = 0.0004

priors = data.frame(alphas = seq(0.1, 250, by = 0.05) )
priors = priors %>%
  mutate(betas = alphas * (1 / mean - 1),
         var = (alphas * betas) / ((alphas + betas)^2  * (alphas + betas + 1) ),
         offtarget = abs(var - target)
         )

  slice(which.min(offtarget) )

# need to find an alpha and beta that simultaneously have alpha/(alpha+beta) = 0.85

mean = 0.85   # set a target for the mean = alpha/(alpha+beta) of our beta distribution
  # set up our probability statement
target = 0.0025

priors = data.frame(alphas = seq(0.1, 250, by = 0.05) )
priors = priors %>%
  mutate(betas = alphas * (1 / mean - 1),
         var = (alphas * betas) / ((alphas + betas)^2  * (alphas + betas + 1) ),
         offtarget = abs(var - target)
         )

priors %>%
  slice(which.min(offtarget) )
